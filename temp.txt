    return generate_cart(data.purchase_history)

✅ utils.py

def generate_recommendations(history):
    recommendations = []

    if "milk" in history:
        recommendations.append("bread")
    if "eggs" in history:
        recommendations.append("butter")
    if "rice" in history:
        recommendations.append("dal")

    return {
        "recommendations": list(set(recommendations)),
        "reason": "Based on frequent co-purchase patterns"
    }


def generate_cart(history):
    cart = history.copy()

    if "milk" in history and "bread" not in history:
        cart.append("bread")

    return {
        "cart": list(set(cart)),
        "confidence": "medium",
        "reason": "Predicted recurring purchase behavior"
    }
